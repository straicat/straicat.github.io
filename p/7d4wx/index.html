public class Stack<T> {
    private class Node{
        T value;
        Node next;
    }
    private Node front;
    private int n;
    public void push(T item) {
        Node tmp = front;
        front = new Node();
        front.value = item;
        front.next = tmp;
        n++;
    }
    public T pop() {
        T tmp = front.value;
        front = front.next;
        n--;
        return tmp;
    }
    public int size() {
        return n;
    }
    public boolean isEmpty() {
        return front == null;
    }
}

public class Queue<T>{
    private class Node{
        T value;
        Node next;
    }
    private Node front;
    private Node rear;
    private int n;
    public void enqueue(T item) {
        Node tmp = rear;
        rear = new Node();
        rear.value = item;
        if(tmp==null)
            front = rear;
        else
            tmp.next = rear;
        n++;
    }
    public T dequeue() {
        T tmp = front.value;
        front = front.next;
        if(front==null)
            rear = null;
        n--;
        return tmp;
    }
    public int size() {
        return n;
    }
    public boolean isEmpty() {
        return front == null;
    }
}
